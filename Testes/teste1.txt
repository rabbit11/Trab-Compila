function fatorial(n : Int) -> Int {
	if n <= 0 {
		return a;
	}
	else {
		return n*fatorial(n-a);
	}
}

function imprima(before: Int, valor: Int, after: String) {
	var i : Int;
	i = 0;
	while i < before {
		write("*");
		i = i + 1;
	}
	writeln("");
	writeln(valor);
	writeln(after);

}

	imprima(50, fatorial(5)*2*fatorial(3), " 	fim");
